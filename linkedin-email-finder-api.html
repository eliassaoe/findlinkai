import requests

url = 'https://api.linkfinderai.com/linkedin_profile_to_linkedin_info'
headers = {
    'Authorization': 'Bearer YOUR_API_KEY',
    'Content-Type': 'application/json'
}
data = {
    'linkedin_url': 'https://linkedin.com/in/john-doe-stripe'
}

response = requests.post(url, headers=headers, json=data)
email_data = response.json()
print(f"Found email: {email_data['email']}")
const fetch = require('node-fetch');

const findEmail = async () => {
  const response = await fetch('https://api.linkfinderai.com/linkedin_profile_to_linkedin_info', {
    method: 'POST',
    headers: {
      'Authorization': 'Bearer YOUR_API_KEY',
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      linkedin_url: 'https://linkedin.com/in/john-doe-stripe'
    })
  });
  
  const emailData = await response.json();
  console.log('Found email:', emailData.email);
};

Frequently asked questions
How accurate is the LinkedIn email finder API?
LinkFinder AI maintains a 99.2% accuracy rate for email discovery from LinkedIn profiles. We use advanced algorithms and real-time verification to ensure data quality.
What data do I need to find emails from LinkedIn?
You only need the LinkedIn profile URL to find associated email addresses. Our API will extract and verify contact information from the provided profile.
Is the LinkedIn email finder API compliant with data regulations?
Yes, LinkFinder AI is fully GDPR compliant and follows all data protection regulations. We only provide publicly available contact information in accordance with LinkedIn's terms.
How fast can I get email results from LinkedIn profiles?
Our API typically responds within 200ms for email lookup requests. Bulk processing is also available for finding emails from multiple LinkedIn profiles simultaneously.
What format does the email finder API return results in?
The API returns results in JSON format including the found email address, confidence score, verification status, and additional profile metadata when available.
