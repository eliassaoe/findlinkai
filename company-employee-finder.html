export default { async fetch(request, env, ctx) { const allowedOrigin = 'https://linkfinderai.com'; // Handle CORS preflight requests if (request.method === 'OPTIONS') { return new Response(null, { headers: { 'Access-Control-Allow-Origin': allowedOrigin, 'Access-Control-Allow-Methods': 'GET, POST, OPTIONS', 'Access-Control-Allow-Headers': 'Content-Type', }, }); } // Check origin const origin = request.headers.get('Origin'); if (origin && origin !== allowedOrigin) { return new Response(JSON.stringify({ error: 'Forbidden', message: 'Access denied from this origin' }), { status: 403, headers: { 'Content-Type': 'application/json', }, }); } try { // Get client IP for rate limiting const clientIP = request.headers.get('CF-Connecting-IP') || request.headers.get('X-Forwarded-For') || 'unknown'; // Parse the incoming request body const requestData = await request.json(); // Validate required parameters - accept companyUrl, linkedin_url, or company_name const companyUrl = requestData.companyUrl || requestData.linkedin_url || requestData.company_url; const companyName = requestData.companyName || requestData.company_name; const maxEmployees = requestData.maxEmployees || requestData.max_employees || "100"; if (!companyUrl && !companyName) { return new Response(JSON.stringify({ error: 'Missing required parameter', message: 'Please provide either companyUrl/linkedin_url or companyName/company_name' }), { status: 400, headers: { 'Content-Type': 'application/json', 'Access-Control-Allow-Origin': allowedOrigin, }, }); } // Validate LinkedIn URL format if provided if (companyUrl && !isValidLinkedInCompanyUrl(companyUrl)) { return new Response(JSON.stringify({ error: 'Invalid LinkedIn Company URL', message: 'Please provide a valid LinkedIn company URL (e.g., https://linkedin.com/company/company-name)' }), { status: 400, headers: { 'Content-Type': 'application/json', 'Access-Control-Allow-Origin': allowedOrigin, }, }); } // Determine if this is a bulk search (more than 50 employees) const isBulkSearch = parseInt(maxEmployees) > 50; // Check rate limits const rateLimitResult = await checkRateLimit(clientIP, isBulkSearch, env); if (!rateLimitResult.allowed) { return new Response(JSON.stringify({ error: 'Rate limit exceeded', message: rateLimitResult.message, requires_signup: true }), { status: 429, headers: { 'Content-Type': 'application/json', 'Access-Control-Allow-Origin': allowedOrigin, 'Retry-After': '3600' // Suggest waiting an hour before trying again }, }); } // Make POST request to the n8n webhook const webhookUrl = 'https://eliasse-n8n.onrender.com/webhook/5d628924-9fc5-4864-be01-8b1731455185'; // Prepare payload for webhook const webhookPayload = { maxEmployees: maxEmployees }; // Add either companyUrl or companyName to payload if (companyUrl) { webhookPayload.companyUrl = companyUrl; } else { webhookPayload.companyName = companyName; } console.log('Sending to n8n webhook:', JSON.stringify(webhookPayload)); const response = await fetch(webhookUrl, { method: 'POST', headers: { 'Content-Type': 'application/json', }, body: JSON.stringify(webhookPayload) }); // Check if the response is ok if (!response.ok) { console.error('Webhook response not ok:', response.status, response.statusText); throw new Error(`Webhook responded with status: ${response.status}`); } // Get the JSON response const data = await response.json(); console.log('Raw n8n response:', JSON.stringify(data)); // Update rate limit counters after successful request await updateRateLimit(clientIP, isBulkSearch, env); console.log('Sending to frontend:', JSON.stringify(data)); // Return the raw data exactly as received from n8n return new Response(JSON.stringify(data), { status: 200, headers: { 'Content-Type': 'application/json', 'Access-Control-Allow-Origin': allowedOrigin, }, }); } catch (error) { console.error('Error in worker:', error); return new Response(JSON.stringify({ error: 'Internal server error', message: error.message }), { status: 500, headers: { 'Content-Type': 'application/json', 'Access-Control-Allow-Origin': allowedOrigin, }, }); } }, }; // Helper function to extract company name from LinkedIn URL function extractCompanyNameFromUrl(url) { if (!url) return null; try { const urlObj = new URL(url); const pathSegments = urlObj.pathname.split('/'); const companyIndex = pathSegments.indexOf('company'); if (companyIndex !== -1 && pathSegments[companyIndex + 1]) { // Convert kebab-case to Title Case return pathSegments[companyIndex + 1] .split('-') .map(word => word.charAt(0).toUpperCase() + word.slice(1)) .join(' '); } } catch (error) { console.error('Error extracting company name from URL:', error); } return null; } // LinkedIn Company URL validation function function isValidLinkedInCompanyUrl(url) { try { const urlObj = new URL(url); const hostname = urlObj.hostname.toLowerCase(); // Check if it's a LinkedIn domain const validDomains = ['linkedin.com', 'www.linkedin.com']; if (!validDomains.includes(hostname)) { return false; } // Check if it's a company URL (contains /company/) const pathname = urlObj.pathname.toLowerCase(); return pathname.includes('/company/'); } catch (error) { return false; } } // Rate limiting functions async function checkRateLimit(clientIP, isBulkSearch, env) { if (!env.RATE_LIMIT_KV) { return { allowed: true }; } const individualKey = `company_individual:${clientIP}`; const bulkKey = `company_bulk:${clientIP}`; try { const [individualCount, bulkCount] = await Promise.all([ env.RATE_LIMIT_KV.get(individualKey), env.RATE_LIMIT_KV.get(bulkKey) ]); const currentIndividualCount = parseInt(individualCount) || 0; const currentBulkCount = parseInt(bulkCount) || 0; if (isBulkSearch) { if (currentBulkCount >= 2) { return { allowed: false, message: 'Free bulk company employees search limit reached (2 bulk searches per user). Please sign up for unlimited access.' }; } } else { if (currentIndividualCount >= 5) { return { allowed: false, message: 'Free company employees search limit reached (5 searches per user). Please sign up for unlimited access.' }; } } return { allowed: true }; } catch (error) { console.error('Rate limit check error:', error); return { allowed: true }; } } async function updateRateLimit(clientIP, isBulkSearch, env) { if (!env.RATE_LIMIT_KV) { return; } const individualKey = `company_individual:${clientIP}`; const bulkKey = `company_bulk:${clientIP}`; try { if (isBulkSearch) { const bulkCount = await env.RATE_LIMIT_KV.get(bulkKey); const newBulkCount = (parseInt(bulkCount) || 0) + 1; await env.RATE_LIMIT_KV.put(bulkKey, newBulkCount.toString(), { expirationTtl: 86400 }); } else { const individualCount = await env.RATE_LIMIT_KV.get(individualKey); const newIndividualCount = (parseInt(individualCount) || 0) + 1; await env.RATE_LIMIT_KV.put(individualKey, newIndividualCount.toString(), { expirationTtl: 86400 }); } } catch (error) { console.error('Rate limit update error:', error); } } async function resetUserLimits(clientIP, env) { if (!env.RATE_LIMIT_KV) { return; } const individualKey = `company_individual:${clientIP}`; const bulkKey = `company_bulk:${clientIP}`; try { await Promise.all([ env.RATE_LIMIT_KV.delete(individualKey), env.RATE_LIMIT_KV.delete(bulkKey) ]); } catch (error) { console.error('Reset limits error:', error); } }
