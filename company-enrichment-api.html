const fetch = require('node-fetch');

const enrichCompany = async (domain) => {
  const response = await fetch('https://api.linkfinderai.com/v1/company-domain-to-employees', {
    method: 'POST',
    headers: {
      'Authorization': 'Bearer YOUR_API_KEY',
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({ domain })
  });
  
  const data = await response.json();
  return data;
};

Frequently asked questions
What company data can I enrich with your API?
Our company enrichment API provides employee counts, company websites, industry classifications, founding dates, headquarters locations, and social media profiles. We also offer technology stack information and recent funding data for supported companies.
How accurate is the company enrichment data?
We maintain 99.9% data accuracy through real-time verification and multiple data sources. Our algorithms cross-reference LinkedIn company pages, official websites, and public databases to ensure information freshness.
Can I enrich companies in bulk or real-time?
Yes, our API supports both batch processing for large datasets and real-time enrichment for individual companies. You can process thousands of companies per minute or integrate real-time enrichment into your application workflows.
What's the difference between your API and other company data providers?
LinkFinder AI specializes in LinkedIn-verified company data with real-time accuracy. Unlike static databases, we provide fresh employee counts, recent company updates, and verified contact information directly sourced from professional networks.
Do you offer company enrichment for international businesses?
Yes, our global database covers companies across 200+ countries and territories. We provide localized data including regional office locations, local employee counts, and country-specific business registration information.
